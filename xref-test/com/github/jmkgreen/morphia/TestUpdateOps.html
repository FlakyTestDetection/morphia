70      public void testIncDec() throws Exception {
71  		Rectangle[] rects = {	new Rectangle(1, 10),
72  								new Rectangle(1, 10),
73  								new Rectangle(1, 10),
74  								new Rectangle(10, 10),
75  								new Rectangle(10, 10),
76  								};
77  		for(Rectangle rect: rects)
78  			ds.save(rect);
79  		
80  		Query<Rectangle> q1 = ds.find(Rectangle.class, "height", 1D);
81  		Query<Rectangle> q2 = ds.find(Rectangle.class, "height", 2D);
82  		
83  		assertEquals(3, ds.getCount(q1));
84  		assertEquals(0, ds.getCount(q2));
85  		
86  		UpdateResults<Rectangle> results = ds.update(q1, ds.createUpdateOperations(Rectangle.class).inc("height"));
87  		assertUpdated(results, 3);
88  		
89  		assertEquals(0, ds.getCount(q1));
90  		assertEquals(3, ds.getCount(q2));
91  
92  		ds.update(q2, ds.createUpdateOperations(Rectangle.class).dec("height"));
93  		assertEquals(3, ds.getCount(q1));
94  		assertEquals(0, ds.getCount(q2));
95  
96  		ds.update(ds.find(Rectangle.class, "width", 1D), ds.createUpdateOperations(Rectangle.class).set("height",1D).set("width", 1D), true);		
97  		assertNotNull(ds.find(Rectangle.class, "width", 1D).get());
98  		assertNull(ds.find(Rectangle.class, "width", 2D).get());
99  		ds.update(ds.find(Rectangle.class, "width", 1D), ds.createUpdateOperations(Rectangle.class).set("height",2D).set("width", 2D), true);		
100 		assertNull(ds.find(Rectangle.class, "width", 1D).get());
101 		assertNotNull(ds.find(Rectangle.class, "width", 2D).get());
102 	}

212 	@Test
213     public void testAdd() throws Exception {
214 		ContainsIntArray cIntArray = new ContainsIntArray();
215 		ds.save(cIntArray);
216 		ContainsIntArray cIALoaded = ds.get(cIntArray);
217 		assertEquals(3, cIALoaded.vals.length);
218 		assertArrayEquals((new ContainsIntArray()).vals, cIALoaded.vals);
219 		
220 		//add 4 to array
221 		UpdateResults<ContainsIntArray> res = ds.updateFirst(ds.createQuery(ContainsIntArray .class), ds.createUpdateOperations(ContainsIntArray.class).add("vals",4, false));
222 		assertUpdated(res, 1);
223 		
224 		cIALoaded = ds.get(cIntArray);
225 		assertArrayEquals(new Integer[]{1,2,3,4}, cIALoaded.vals);
226 
227 		//add unique (4) -- noop
228 		res = ds.updateFirst(ds.createQuery(ContainsIntArray.class), ds.createUpdateOperations(ContainsIntArray.class).add("vals",4, false));
229 		assertUpdated(res, 1);
230 		
231 		cIALoaded = ds.get(cIntArray);
232 		assertArrayEquals(new Integer[]{1,2,3,4}, cIALoaded.vals);
233 
234 		//add dup 4
235 		res = ds.updateFirst(ds.createQuery(ContainsIntArray.class), ds.createUpdateOperations(ContainsIntArray.class).add("vals",4, true));
236 		assertUpdated(res, 1);
237 		
238 		cIALoaded = ds.get(cIntArray);
239 		assertArrayEquals(new Integer[]{1,2,3,4,4}, cIALoaded.vals);
240 
241 		//cleanup for next tests
242 		ds.delete(ds.find(ContainsIntArray.class));
243 		cIntArray = ds.getByKey(ContainsIntArray.class, ds.save(new ContainsIntArray()));
244 		
245 		//add [4,5]
246 		List<Integer> newVals = new ArrayList<Integer>();
247 		newVals.add(4);newVals.add(5);
248 		res = ds.updateFirst(ds.createQuery(ContainsIntArray.class), ds.createUpdateOperations(ContainsIntArray.class).addAll("vals", newVals, false));
249 		assertUpdated(res, 1);
250 		
251 		cIALoaded = ds.get(cIntArray);
252 		assertArrayEquals(new Integer[]{1,2,3,4,5}, cIALoaded.vals);
253 		
254 		//add them again... noop
255 		res = ds.updateFirst(ds.createQuery(ContainsIntArray.class), ds.createUpdateOperations(ContainsIntArray.class).addAll("vals", newVals, false));
256 		assertUpdated(res, 1);
257 		
258 		cIALoaded = ds.get(cIntArray);
259 		assertArrayEquals(new Integer[]{1,2,3,4,5}, cIALoaded.vals);
260 
261 		//add dups [4,5]
262 		res = ds.updateFirst(ds.createQuery(ContainsIntArray.class), ds.createUpdateOperations(ContainsIntArray.class).addAll("vals", newVals, true));
263 		assertUpdated(res, 1);
264 		
265 		cIALoaded = ds.get(cIntArray);
266 		assertArrayEquals(new Integer[]{1,2,3,4,5,4,5}, cIALoaded.vals);
