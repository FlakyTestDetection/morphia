39  public class IgnoreFieldsAnnotationTest extends TestBase {
40  
41  	public IgnoreFieldsAnnotationTest () {
42  		super();
43  	}
44  
45  	Transient transAnn;
46  	@Retention(RetentionPolicy.RUNTIME)
47  	@Target({ElementType.TYPE})
48  	static @interface IgnoreFields {
49  		String value();
50  	}
51  
52  	@Entity
53  	@IgnoreFields("ignored")
54  	static class User {
55  		@Id ObjectId id;
56  		String email;
57  		String ignored = "never, never";
58  	}
59  
60  	@Before
61  	public void setUp() {
62  		super.setUp();
63  		MappedClass.interestingAnnotations.add(IgnoreFields.class);
64  		this.morphia.map(User.class);
65  		processIgnoreFieldsAnnotations();
66  	}
67  
68  	//remove any MappedField specified in @IngoreFields on the class.
69  	void processIgnoreFieldsAnnotations(){
70  		for(MappedClass mc : ds.getMapper().getMappedClasses()) {
71  			IgnoreFields ignores = (IgnoreFields) mc.getAnnotation(IgnoreFields.class);
72  			if (ignores != null) {
73  				for(String field : ignores.value().split(",")) {
74  					MappedField mf = mc.getMappedFieldByJavaField(field);
75  					mc.getMappedFields().remove(mf);
76  				}
77  			}
78  		}
79  	}
80  	@Test
81  	public void testIt() {
82  		User u = new User();
83  		u.email = "ScottHernandez@gmail.com";
84  		u.ignored = "test";
85  		ds.save(u);
86  
87  		User uLoaded = ds.find(User.class).get();
88  		Assert.assertEquals("never, never", uLoaded.ignored);
89  	}
