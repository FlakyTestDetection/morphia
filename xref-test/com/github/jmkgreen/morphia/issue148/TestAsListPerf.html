223 	public static double morphiaQueryAndMorphiaConv(final int nbOfHits, final Datastore ds, final Morphia morphia) {
224 		Query<Address> query = ds.createQuery(Address.class).
225 									order("name");
226 		long start = System.nanoTime();
227 		List<Address> resultList = query.asList();
228 		long duration = (System.nanoTime() - start) / 1000000; //ns -> ms
229 		Assert.assertEquals(nbOfHits, resultList.size());
230 		return (double)duration/nbOfHits;
231 	}

233 	public static double driverQueryAndMorphiaConv(final int nbOfHits, final Datastore ds, final Morphia morphia) {
234 		long start = System.nanoTime();
235 		List<DBObject> list = ds.getDB().getCollection("Address").
236 									find().
237 									sort(new BasicDBObject("name", 1)).
238 									toArray();
239 		EntityCache entityCache = new DefaultEntityCache();
240 		List<Address> resultList = new LinkedList<Address>();
241 		for (DBObject dbObject : list) {
242 			Address address = morphia.fromDBObject(Address.class, dbObject, entityCache);
243 			resultList.add(address);
244 		}
245 		long duration = (System.nanoTime() - start) / 1000000; //ns -> ms
246 		Assert.assertEquals(nbOfHits, resultList.size());
247 		return (double)duration/nbOfHits;
248 	}
