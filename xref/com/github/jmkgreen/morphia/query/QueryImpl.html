144     public DBObject getFieldsObject() {
145         if (fields == null || fields.length == 0)
146             return null;
147 
148         Map<String, Integer> fieldsFilter = new HashMap<String, Integer>();
149         for (String field : this.fields) {
150             StringBuffer sb = new StringBuffer(field); //validate might modify prop string to translate java field name to db field name
151             DefaultMapper.validate(clazz, ds.getMapper(), sb, FilterOperator.EQUAL, null, validateName, false);
152             field = sb.toString();
153             fieldsFilter.put(field, (includeFields ? 1 : 0));
154         }
155 
156         //Add className field just in case.
157         if (includeFields)
158             fieldsFilter.put(Mapper.CLASS_NAME_FIELDNAME, 1);
159 
160         return new BasicDBObject(fieldsFilter);
161     }

182     public DBCursor prepareCursor() {
183         DBObject query = getQueryObject();
184         DBObject fields = getFieldsObject();
185 
186         if (log.isTraceEnabled())
187             log.trace("Running query(" + dbColl.getName() + ") : " + query + ", fields:" + fields + ",off:" + offset + ",limit:" + limit);
188 
189         DBCursor cursor = dbColl.find(query, fields);
190         cursor.setDecoderFactory(this.ds.getDecoderFact());
191 
192         if (offset > 0)
193             cursor.skip(offset);
194         if (limit > 0)
195             cursor.limit(limit);
196         if (batchSize > 0)
197             cursor.batchSize(batchSize);
198         if (snapshotted)
199             cursor.snapshot();
200         if (sort != null)
201             cursor.sort(sort);
202         if (indexHint != null)
203             cursor.hint(indexHint);
204 
205         if (null != readPref) {
206             cursor.setReadPreference(readPref);
207         }
208 
209         if (noTimeout) {
210             cursor.addOption(Bytes.QUERYOPTION_NOTIMEOUT);
211         }
212 
213         if (tail) {
214             cursor.addOption(Bytes.QUERYOPTION_TAILABLE);
215             if (tail_await_data)
216                 cursor.addOption(Bytes.QUERYOPTION_AWAITDATA);
217         }
218 
219         //Check for bad options.
220         if (snapshotted && (sort != null || indexHint != null))
221             log.warning("Snapshotted query should not have hint/sort.");
222 
223         if (tail && (sort != null)) {
224             // i don´t think that just warning is enough here, i´d favor a RTE, agree?
225             log.warning("Sorting on tail is not allowed.");
226         }
227 
228         return cursor;
229     }

296     /**
297      * Converts the textual operator (">", "<=", etc) into a FilterOperator.
298      * Forgiving about the syntax; != and <> are NOT_EQUAL, = and == are EQUAL.
299      */
300     protected FilterOperator translate(String operator) {
301         operator = operator.trim();
302 
303         if (operator.equals("=") || operator.equals("=="))
304             return FilterOperator.EQUAL;
305         else if (operator.equals(">"))
306             return FilterOperator.GREATER_THAN;
307         else if (operator.equals(">="))
308             return FilterOperator.GREATER_THAN_OR_EQUAL;
309         else if (operator.equals("<"))
310             return FilterOperator.LESS_THAN;
311         else if (operator.equals("<="))
312             return FilterOperator.LESS_THAN_OR_EQUAL;
313         else if (operator.equals("!=") || operator.equals("<>"))
314             return FilterOperator.NOT_EQUAL;
315         else if (operator.toLowerCase().equals("in"))
316             return FilterOperator.IN;
317         else if (operator.toLowerCase().equals("nin"))
318             return FilterOperator.NOT_IN;
319         else if (operator.toLowerCase().equals("all"))
320             return FilterOperator.ALL;
321         else if (operator.toLowerCase().equals("exists"))
322             return FilterOperator.EXISTS;
323         else if (operator.toLowerCase().equals("elem"))
324             return FilterOperator.ELEMENT_MATCH;
325         else if (operator.toLowerCase().equals("size"))
326             return FilterOperator.SIZE;
327         else if (operator.toLowerCase().equals("within"))
328             return FilterOperator.WITHIN;
329         else if (operator.toLowerCase().equals("near"))
330             return FilterOperator.NEAR;
331         else
332             throw new IllegalArgumentException("Unknown operator '" + operator + "'");
333     }

335     public Query<T> filter(String condition, Object value) {
336         String[] parts = condition.trim().split(" ");
337         if (parts.length < 1 || parts.length > 6)
338             throw new IllegalArgumentException("'" + condition + "' is not a legal filter condition");
339 
340         String prop = parts[0].trim();
341         FilterOperator op = (parts.length == 2) ? this.translate(parts[1]) : FilterOperator.EQUAL;
342 
343         this.add(new FieldCriteria(this, prop, op, value, this.validateName, this.validateType));
344 
345         return this;
346     }

434     /**
435      * parses the string and validates each part
436      */
437     @SuppressWarnings("rawtypes")
438     public static BasicDBObject parseFieldsString(String str, Class clazz, Mapper mapr, boolean validate) {
439         BasicDBObjectBuilder ret = BasicDBObjectBuilder.start();
440         String[] parts = str.split(",");
441         for (String s : parts) {
442             s = s.trim();
443             int dir = 1;
444 
445             if (s.startsWith("-")) {
446                 dir = -1;
447                 s = s.substring(1).trim();
448             }
449 
450             if (validate) {
451                 StringBuffer sb = new StringBuffer(s);
452                 DefaultMapper.validate(clazz, mapr, sb, FilterOperator.IN, "", true, false);
453                 s = sb.toString();
454             }
455             ret = ret.add(s, dir);
456         }
457         return (BasicDBObject) ret.get();
458     }
