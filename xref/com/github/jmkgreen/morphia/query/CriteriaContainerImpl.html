38      public void addTo(DBObject obj) {
39          if (this.joinMethod == CriteriaJoin.AND) {
40              Set<String> fields = new HashSet<String>();
41              int nonNullFieldNames = 0;
42              for (Criteria child : this.children) {
43                  if (null != child.getFieldName()) {
44                      fields.add(child.getFieldName());
45                      nonNullFieldNames++;
46                  }
47              }
48              if (fields.size() < nonNullFieldNames) {
49                  //use $and
50                  BasicDBList and = new BasicDBList();
51  
52                  for (Criteria child : this.children) {
53                      BasicDBObject container = new BasicDBObject();
54                      child.addTo(container);
55                      and.add(container);
56                  }
57  
58                  obj.put("$and", and);
59              } else {
60                  //no dup field names, don't use $and
61                  for (Criteria child : this.children) {
62                      child.addTo(obj);
63                  }
64              }
65          } else if (this.joinMethod == CriteriaJoin.OR) {
66              BasicDBList or = new BasicDBList();
67  
68              for (Criteria child : this.children) {
69                  BasicDBObject container = new BasicDBObject();
70                  child.addTo(container);
71                  or.add(container);
72              }
73  
74              obj.put("$or", or);
75          }
76      }
