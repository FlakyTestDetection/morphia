21      public ContradictingFieldAnnotation(Class<? extends Annotation> a1, Class<? extends Annotation> a2) {

24      }

26      @Override
27      protected final void check(MappedClass mc, MappedField mf, Set<ConstraintViolation> ve) {
28          if (mf.hasAnnotation(a1) && mf.hasAnnotation(a2))
29              ve.add(new ConstraintViolation(Level.FATAL, mc, mf, this.getClass(), "A field can be either annotated with @"
