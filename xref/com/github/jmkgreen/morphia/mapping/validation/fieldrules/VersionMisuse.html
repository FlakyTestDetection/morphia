17  public class VersionMisuse extends FieldConstraint {
18  
19      @Override
20      protected void check(MappedClass mc, MappedField mf, Set<ConstraintViolation> ve) {
21          if (mf.hasAnnotation(Version.class)) {
22              Class<?> type = mf.getType();
23              if (Long.class.equals(type) || long.class.equals(type)) {
24  
25                  //TODO: Replace this will a read ObjectFactory call -- requires Mapper instance.
26                  Object testInstance = DefaultCreator.createInst(mc.getClazz());
27  
28                  // check initial value
29                  if (Long.class.equals(type)) {
30                      if (mf.getFieldValue(testInstance) != null)
31                          ve.add(new ConstraintViolation(Level.FATAL, mc, mf, this.getClass(), "When using @" + Version.class.getSimpleName()
32                                  + " on a Long field, it must be initialized to null."));
33                  } else if (long.class.equals(type)) {
34                      if ((Long) mf.getFieldValue(testInstance) != 0L)
35                          ve.add(new ConstraintViolation(Level.FATAL, mc, mf, this.getClass(), "When using @" + Version.class.getSimpleName()
36                                  + " on a long field, it must be initialized to 0."));
37                  }
38              } else
39                  ve.add(new ConstraintViolation(Level.FATAL, mc, mf, this.getClass(), "@" + Version.class.getSimpleName() + " can only be used on a Long/long field."));
40          }
41      }
