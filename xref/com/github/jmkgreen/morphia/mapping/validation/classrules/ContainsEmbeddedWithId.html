18  public class ContainsEmbeddedWithId implements ClassConstraint {

20      private boolean hasTypeFieldAnnotation(final Class<?> type, final Class<Id> class1) {
21          for (Field field : ReflectionUtils.getDeclaredAndInheritedFields(type, true)) {
22              if (field.getAnnotation(class1) != null) {
23                  return true;
24              }
25          }
26          return false;
27      }
28  
29      public void check(final MappedClass mc, final Set<ConstraintViolation> ve) {
30          Set<Class<?>> classesToInspect = new HashSet<Class<?>>();
31          for (Field field : ReflectionUtils.getDeclaredAndInheritedFields(mc.getClazz(), true)) {
32              if (isFieldToInspect(field) && !field.isAnnotationPresent(Id.class)) {
33                  classesToInspect.add(field.getType());
34              }
35          }
36          checkRecursivelyHasNoIdAnnotationPresent(classesToInspect, new HashSet<Class<?>>(), mc, ve);
37      }
38  
39      private boolean isFieldToInspect(final Field field) {
40          return (!field.isAnnotationPresent(Transient.class) && !field.isAnnotationPresent(Reference.class));
41      }
42  
43      private void checkRecursivelyHasNoIdAnnotationPresent(final Set<Class<?>> classesToInspect,
44                                                            final HashSet<Class<?>> alreadyInspectedClasses, final MappedClass mc, final Set<ConstraintViolation> ve) {
45          for (Class<?> clazz : classesToInspect) {
46              if (alreadyInspectedClasses.contains(clazz)) {
47                  continue;
48              }
49              if (hasTypeFieldAnnotation(clazz, Id.class)) {
50                  ve.add(new ConstraintViolation(Level.FATAL, mc, this.getClass(),
51                          "You cannot use @Id on any field of an Embedded/Property object"));
52              }
53              alreadyInspectedClasses.add(clazz);
54              Set<Class<?>> extraClassesToInspect = new HashSet<Class<?>>();
55              for (Field field : ReflectionUtils.getDeclaredAndInheritedFields(clazz, true)) {
56                  if (isFieldToInspect(field)) {
57                      extraClassesToInspect.add(field.getType());
58                  }
59              }
60              checkRecursivelyHasNoIdAnnotationPresent(extraClassesToInspect, alreadyInspectedClasses, mc, ve);
61          }
62      }
