135     protected void basicValidate() {
136         boolean isStatic = Modifier.isStatic(clazz.getModifiers());
137         if (!isStatic && clazz.isMemberClass())
138             throw new MappingException("Cannot use non-static inner class: " + clazz + ". Please make static.");

156     /**
157      * Discovers interesting (that we care about) things about the class.
158      */
159     protected void discover() {
160         for (Class<? extends Annotation> c : interestingAnnotations) {
161             addAnnotation(c);
162         }
163 
164         List<Class<?>> lifecycleClasses = new ArrayList<Class<?>>();
165         lifecycleClasses.add(clazz);
166 
167         EntityListeners entityLisAnn = (EntityListeners) getAnnotation(EntityListeners.class);
168         if (entityLisAnn != null && entityLisAnn.value() != null && entityLisAnn.value().length != 0)
169             for (Class<?> c : entityLisAnn.value())
170                 lifecycleClasses.add(c);
171 
172         for (Class<?> cls : lifecycleClasses) {
173             for (Method m : ReflectionUtils.getDeclaredAndInheritedMethods(cls)) {
174                 for (Class<? extends Annotation> c : lifecycleAnnotations) {
175                     if (m.isAnnotationPresent(c)) {
176                         addLifecycleEventMethod(c, m, cls.equals(clazz) ? null : cls);
177                     }
178                 }
179             }
180         }
181 
182         update();
183 
184         for (Field field : ReflectionUtils.getDeclaredAndInheritedFields(clazz, true)) {
185             field.setAccessible(true);
186             int fieldMods = field.getModifiers();
187             if (field.isAnnotationPresent(Transient.class))
188                 continue;
189             else if (field.isSynthetic() && (fieldMods & Modifier.TRANSIENT) == Modifier.TRANSIENT)
190                 continue;
191             else if (mapr.getOptions().actLikeSerializer && ((fieldMods & Modifier.TRANSIENT) == Modifier.TRANSIENT))
192                 continue;
193             else if (mapr.getOptions().ignoreFinals && ((fieldMods & Modifier.FINAL) == Modifier.FINAL))
194                 continue;
195             else if (field.isAnnotationPresent(Id.class)) {
196                 MappedField mf = new MappedField(field, clazz);
197                 mappedFields.add(mf);
198                 update();
199             } else if (field.isAnnotationPresent(Property.class) ||
200                     field.isAnnotationPresent(Reference.class) ||
201                     field.isAnnotationPresent(Embedded.class) ||
202                     field.isAnnotationPresent(Serialized.class) ||
203                     isSupportedType(field.getType()) ||
204                     ReflectionUtils.implementsInterface(field.getType(), Serializable.class)) {
205                 mappedFields.add(new MappedField(field, clazz));
206             } else {
207                 if (mapr.getOptions().defaultMapper != null)
208                     mappedFields.add(new MappedField(field, clazz));
209                 else if (log.isWarningEnabled())
210                     log.warning("Ignoring (will not persist) field: " + clazz.getName() + "." + field.getName() + " [type:" + field.getType().getName() + "]");
211             }
212         }
213     }

215     private void addLifecycleEventMethod(Class<? extends Annotation> lceClazz, Method m, Class<?> clazz) {
216         ClassMethodPair cm = new ClassMethodPair(clazz, m);
217         if (lifecycleMethods.containsKey(lceClazz))
218             lifecycleMethods.get(lceClazz).add(cm);
219         else {
220             ArrayList<ClassMethodPair> methods = new ArrayList<ClassMethodPair>();
221             methods.add(cm);
222             lifecycleMethods.put(lceClazz, methods);
223         }
224     }

232     public void addAnnotation(Class<? extends Annotation> clazz, Annotation ann) {
233         if (ann == null || clazz == null) return;

235         if (!this.foundAnnotations.containsKey(clazz)) {
236             ArrayList<Annotation> list = new ArrayList<Annotation>();
237             foundAnnotations.put(clazz, list);
238         }
239 
240         foundAnnotations.get(clazz).add(ann);

305     /**
306      * Checks to see if it a Map/Set/List or a property supported by the MangoDB java driver
307      */
308     public static boolean isSupportedType(Class<?> clazz) {
309         if (ReflectionUtils.isPropertyType(clazz)) return true;
310         if (clazz.isArray() || Map.class.isAssignableFrom(clazz) || Iterable.class.isAssignableFrom(clazz)) {
311             Class<?> subType = null;
312             if (clazz.isArray()) subType = clazz.getComponentType();
313             else subType = ReflectionUtils.getParameterizedClass(clazz);
314 
315             //get component type, String.class from List<String>
316             if (subType != null && subType != Object.class && !ReflectionUtils.isPropertyType(subType))
317                 return false;
318 
319             //either no componentType or it is an allowed type
320             return true;
321         }
322         return false;
323     }

350     /**
351      * Call the lifecycle methods on the
352      */
353     public DBObject callLifecycleMethods(Class<? extends Annotation> event, Object entity, DBObject dbObj, Mapper mapr) {
354         List<ClassMethodPair> methodPairs = getLifecycleMethods((Class<Annotation>) event);
355         DBObject retDbObj = dbObj;
356         try {
357             Object tempObj = null;
358             if (methodPairs != null) {
359                 HashMap<Class<?>, Object> toCall = new HashMap<Class<?>, Object>((int) (methodPairs.size() * 1.3));
360                 for (ClassMethodPair cm : methodPairs)
361                     toCall.put(cm.clazz, null);
362                 for (Class<?> c : toCall.keySet())
363                     if (c != null)
364                         toCall.put(c, getOrCreateInstance(c));
365 
366                 for (ClassMethodPair cm : methodPairs) {
367                     Method method = cm.method;
368                     Class<?> type = cm.clazz;
369 
370                     Object inst = toCall.get(type);
371                     method.setAccessible(true);
372 
373                     if (log.isDebugEnabled())
374                         log.debug("Calling lifecycle method(@" + event.getSimpleName() + " " + method + ") on " + inst + "");
375 
376                     if (inst == null)
377                         if (method.getParameterTypes().length == 0)
378                             tempObj = method.invoke(entity);
379                         else
380                             tempObj = method.invoke(entity, retDbObj);
381                     else if (method.getParameterTypes().length == 0)
382                         tempObj = method.invoke(inst);
383                     else if (method.getParameterTypes().length == 1)
384                         tempObj = method.invoke(inst, entity);
385                     else
386                         tempObj = method.invoke(inst, entity, retDbObj);
387 
388                     if (tempObj != null)
389                         retDbObj = (DBObject) tempObj;
390                 }
391             }
392 
393             callGlobalInterceptors(event, entity, dbObj, mapr, mapr.getInterceptors());
394         } catch (IllegalAccessException e) {
395             throw new RuntimeException(e);
396         } catch (InvocationTargetException e) {
397             throw new RuntimeException(e);
398         }
399 
400         return retDbObj;
401     }

403     private Object getOrCreateInstance(Class<?> clazz) {
404         if (mapr.isCached(clazz))
405             return mapr.getCachedClass(clazz);
406 
407         Object o = mapr.getOptions().objectFactory.createInstance(clazz);
408         try {
409         	mapr.cacheClass(clazz, o);
410         } catch (ConcurrentModificationException duplicate) {
411             if (log.isErrorEnabled())
412                 log.error("Race-condition",duplicate);
413         }
414         return o;
415 
416     }

418     private void callGlobalInterceptors(Class<? extends Annotation> event, Object entity, DBObject dbObj, Mapper mapr,
419                                         Collection<EntityInterceptor> interceptors) {
420         for (EntityInterceptor ei : interceptors) {
421             if (log.isDebugEnabled())
422                 log.debug("Calling interceptor method " + event.getSimpleName() + " on " + ei);
423 
424             if (event.equals(PreLoad.class)) ei.preLoad(entity, dbObj, mapr);
425             else if (event.equals(PostLoad.class)) ei.postLoad(entity, dbObj, mapr);
426             else if (event.equals(PrePersist.class)) ei.prePersist(entity, dbObj, mapr);
427             else if (event.equals(PreSave.class)) ei.preSave(entity, dbObj, mapr);
428             else if (event.equals(PostPersist.class)) ei.postPersist(entity, dbObj, mapr);
429         }
