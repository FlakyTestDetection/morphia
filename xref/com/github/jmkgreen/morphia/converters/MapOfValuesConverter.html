17  @SuppressWarnings({"unchecked", "rawtypes"})
18  public class MapOfValuesConverter extends TypeConverter {
19      private final DefaultConverters converters;
20  
21      public MapOfValuesConverter(DefaultConverters converters) {
22          this.converters = converters;
23      }
24  
25      @Override
26      protected boolean isSupported(Class<?> c, MappedField optionalExtraInfo) {
27          if (optionalExtraInfo != null)
28              return optionalExtraInfo.isMap();
29          else
30              return ReflectionUtils.implementsInterface(c, Map.class);
31      }
32  
33      @Override
34      public Object decode(Class targetClass, Object fromDBObject, final MappedField mf) throws MappingException {
35          if (fromDBObject == null) return null;
36  
37  
38          final Map values = mapr.getOptions().objectFactory.createMap(mf);
39          new IterHelper<Object, Object>().loopMap((Object) fromDBObject, new MapIterCallback<Object, Object>() {
40              @Override
41              public void eval(Object key, Object val) {
42                  Object objKey = converters.decode(mf.getMapKeyClass(), key);
43                  values.put(objKey, converters.decode(mf.getSubClass(), val));
44              }
45          });
46  
47          return values;
48      }
49  
50      @Override
51      public Object encode(Object value, MappedField mf) {
52          if (value == null)
53              return null;
54  
55          Map<Object, Object> map = (Map<Object, Object>) value;
56          if ((map != null) && (map.size() > 0)) {
57              Map mapForDb = new HashMap();
58              for (Map.Entry<Object, Object> entry : map.entrySet()) {
59                  String strKey = converters.encode(entry.getKey()).toString();
60                  mapForDb.put(strKey, converters.encode(entry.getValue()));
61              }
62              return mapForDb;
63          }
64          return null;
65      }
