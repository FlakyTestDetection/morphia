21  @SuppressWarnings({"unchecked", "rawtypes"})
22  public class IterableConverter extends TypeConverter {
23      private final DefaultConverters chain;
24  
25      public IterableConverter(DefaultConverters chain) {
26          this.chain = chain;
27      }
28  
29      @Override
30      protected boolean isSupported(Class c, MappedField mf) {
31          if (mf != null)
32              return mf.isMultipleValues() && !mf.isMap(); //&& !mf.isTypeMongoCompatible();
33          else
34              return c.isArray() || ReflectionUtils.implementsInterface(c, Iterable.class);
35      }
36  
37      @Override
38      public Object decode(Class targetClass, Object fromDBObject, MappedField mf) throws MappingException {
39          if (mf == null || fromDBObject == null) return fromDBObject;
40  
41          Class subtypeDest = mf.getSubClass();
42          Collection vals = createNewCollection(mf);
43  
44          if (fromDBObject.getClass().isArray()) {
45              //This should never happen. The driver always returns list/arrays as a List
46              for (Object o : (Object[]) fromDBObject)
47                  vals.add(chain.decode((subtypeDest != null) ? subtypeDest : o.getClass(), o));
48          } else if (fromDBObject instanceof Iterable) {
49              // map back to the java datatype
50              // (List/Set/Array[])
51              for (Object o : (Iterable) fromDBObject)
52                  vals.add(chain.decode((subtypeDest != null) ? subtypeDest : o.getClass(), o));
53          } else {
54              //Single value case.
55              vals.add(chain.decode((subtypeDest != null) ? subtypeDest : fromDBObject.getClass(), fromDBObject));
56          }
57  
58          //convert to and array if that is the destination type (not a list/set)
59          if (mf.getType().isArray()) {
60              return ReflectionUtils.convertToArray(subtypeDest, (ArrayList) vals);
61          } else
62              return vals;
63      }
64  
65      private Collection<?> createNewCollection(final MappedField mf) {
66          ObjectFactory of = mapr.getOptions().objectFactory;
67          return mf.isSet() ? of.createSet(mf) : of.createList(mf);
68      }
69  
70      @Override
71      public Object encode(Object value, MappedField mf) {
72  
73          if (value == null)
74              return null;
75  
76          Iterable<?> iterableValues = null;
77  
78          if (value.getClass().isArray()) {
79  
80              if (Array.getLength(value) == 0) {
81                  return value;
82              }
83  
84              if (value.getClass().getComponentType().isPrimitive())
85                  return value;
86  
87              iterableValues = Arrays.asList((Object[]) value);
88          } else {
89              if (!(value instanceof Iterable))
90                  throw new ConverterException("Cannot cast " + value.getClass() + " to Iterable for MappedField: " + mf);
91  
92              // cast value to a common interface
93              iterableValues = (Iterable<?>) value;
94          }
95  
96          List values = new ArrayList();
97          if (mf != null && mf.getSubClass() != null) {
98              for (Object o : iterableValues) {
99                  values.add(chain.encode(mf.getSubClass(), o));
100             }
101         } else {
102             for (Object o : iterableValues) {
103                 values.add(chain.encode(o));
104             }
105         }
106         if (values.size() > 0 || mapr.getOptions().storeEmpties) {
107             return values;
108         } else
109             return null;
110     }
